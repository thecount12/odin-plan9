#include "syscalls.h"
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// File operations
int odin_open(const char *path, int flags) {
    return open(path, flags);
}

ssize_t odin_read(int fd, void *buf, size_t count) {
    return read(fd, buf, count);
}

ssize_t odin_write(int fd, const void *buf, size_t count) {
    return write(fd, buf, count);
}

// Process management
pid_t odin_fork(void) {
    return fork();
}

int odin_execve(const char *filename, char *const argv[], char *const envp[]) {
    return execve(filename, argv, envp);
}

// Memory management
void *odin_malloc(size_t size) {
    return malloc(size);
}

void odin_free(void *ptr) {
    free(ptr);
}
